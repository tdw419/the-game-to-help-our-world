            const indicator = document.getElementById('covenantIndicator');
            if (safe) {
                indicator.classList.remove('alert');
                setTimeout(() => {

                }, 2000);
            } else {
                indicator.classList.add('alert');
                setTimeout(() => {
                    indicator.classList.remove('alert');
                }, 3000);
            }
